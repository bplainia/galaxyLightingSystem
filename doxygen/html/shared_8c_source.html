185 unsigned i2c_rx(unsigned char addr, unsigned char reg16, unsigned char regl, unsigned char regh, unsigned char *data, unsigned int dataLength) // recieve data from address
186 {
187  unsigned char i = 0;
188  if(dataLength == 0) return true; // Do nothing if there is a zero data length.
189 
190  // write address
191  i2c_start();
192  i2c_send(addr & 0b11111110); // write mode at address
193  if(reg16) i2c_send(regh); // Address High Byte
194  i2c_send(regl); // Address Low Byte
195  i2c_restart(); // exit write mode
196  i2c_send(addr | 0b00000001); // read mode at address
197  do
198  {
199  data[i] = i2c_recv(0); // Ack
200  }
201  while (++i < dataLength);
202  data[i] = i2c_recv(1); // NOAck
203  i2c_stop();
204  return false;
205 }
