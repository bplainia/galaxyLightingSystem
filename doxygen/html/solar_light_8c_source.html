188 void interrupt isr()
189 {
190  volatile unsigned char temp, temp2;
191  if(RC1IE && RC1IF)
192  {
193  RC1IF = 0; // reset the flag
194  temp2 = RCSTA1bits.RX9D;
195  temp = RCREG1;
196 
197  // If we recieve master active, comm state machine goes to slave mode
198  if(COMSTAT.STATE == 0b000) // there is no master and I am not the master
199  {
200  if(temp2 == 1 && temp == 254) // if was master active address...
201  {
202  if(masterAddr!=250) // check to see if master address is set.
203  {
204  COMSTAT.TERROR = 1; // token error if it was already
205  }
206  }
207  else
208  {
209  COMSTAT.STATE = 0b001;
210  RCSTA1bits.ADDEN = 1; // back to address mode
211  }
212  }
213  else if(COMSTAT.STATE == 0b001) // Slave is idle
214  {
215  if(temp == 253) // master now inactive. Bye Maintainence Man.
216  {
217  COMSTAT.STATE = 0b000; // inactive bus
218  masterAddr = 250; // reset to global address
219  RCSTA1bits.ADDEN = 1;
220  }
221  else if(temp == myAddr)
222  {
223  COMSTAT.STATE = 0b010; // now listening since it is my address
224  // no address mode for this since we are waiting for data.
225  }
226  else if(temp == 250)
227  {
228  COMSTAT.STATE = 0b111;
229  // global address. waiting for data...
230  }
231  }
232  else if(COMSTAT.STATE == 0b010 || COMSTAT.STATE == 0b111) // Slave is listening state
233  {
234  if(temp2 == 0)
235  {
236  rxBuff[rxPtrIn++] = temp;
237  if(rxPtrIn >= RXBUFFSIZE) rxPtrIn = 0;
238  if(rxPtrIn == rxPtrOut) COMSTAT.RXOVER = 1; // overflow
239  }
240  else if(temp2 == 1) // My turn to execute!
241  {
242  if(temp == 252) // if slave active byte
243  {
244  COMSTAT.STATE = 0b011; // slave talking now
245  }
246  else
247  {
248  COMSTAT.TERROR = 1;
249  comm_flush(); // clear the command buffer
250  COMSTAT.STATE = 0b001; // back to just listening
251  }
252  }
253  }
254  } // end of uart1 recieve
255 
256  else if(RC3IE && RC3IF) // uart3 recieve (keypad entry)
257  {
258  temp = RCREG3; // get the byte just sent from the ede1144
259  RC3IF = 0; // reset the flag
260  temp = 10;
261  switch(temp)
