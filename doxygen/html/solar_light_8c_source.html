198 void interrupt low_priority isr()
199 {
200  volatile unsigned char temp, temp2;
201  if(RC1IE && RC1IF)
202  {
203  RC1IF = 0; // reset the flag
204  temp2 = RCSTA1bits.RX9D;
205  temp = RCREG1;
206 
207  // If we recieve master active, comm state machine goes to slave mode
208  if(COMSTAT.STATE == 0b000) // there is no master and I am not the master
209  {
210  if(temp2 == 1 && temp == 254) // if was master active address...
211  {
212  if(masterAddr!=250) // check to see if master address is set.
213  {
214  COMSTAT.TERROR = 1; // token error if it was already
215  }
216  }
217  else
218  {
219  masterAddr = temp;
220  COMSTAT.STATE = 0b001;
221  status.mmode = 0b10; // slave mode
222  RCSTA1bits.ADDEN = 1; // back to address mode
223  }
224  }
225  else if(COMSTAT.STATE == 0b001) // Slave is idle
226  {
227  if(temp == 253) // master now inactive. Bye Maintainence Man.
228  {
229  COMSTAT.STATE = 0b000; // inactive bus
230  masterAddr = 250; // reset to global address
231  RCSTA1bits.ADDEN = 1;
232  }
233  else if(temp == myAddr)
234  {
235  COMSTAT.STATE = 0b010; // now listening since it is my address
236  // no address mode for this since we are waiting for data.
237  }
238  else if(temp == 250)
239  {
240  COMSTAT.STATE = 0b111;
241  // global address. waiting for data...
242  }
243  }
244  else if(COMSTAT.STATE == 0b010 || COMSTAT.STATE == 0b111) // Slave is listening state
245  {
246  if(temp2 == 0)
247  {
248  rxBuff[rxPtrIn++] = temp;
249  if(rxPtrIn >= RXBUFFSIZE) rxPtrIn = 0;
250  if(rxPtrIn == rxPtrOut) COMSTAT.RXOVER = 1; // overflow
251  }
252  else if(temp2 == 1) // My turn to execute!
253  {
254  if(temp == 252) // if slave active byte
255  {
256  COMSTAT.STATE = 0b011; // slave talking now
257  }
258  else
259  {
260  COMSTAT.TERROR = 1;
261  comm_flush(); // clear the command buffer
262  COMSTAT.STATE = 0b001; // back to just listening
263  }
264  }
265  }
266  } // end of uart1 recieve
267 
268  else if(RC3IE && RC3IF) // uart3 recieve (keypad entry)
269  {
270  keypad_push(RCREG3);
271  RC3IF = 0; // reset the flag
272  }
273 }
