92 /*********************************************************************
93 * Function: void APP_DeviceCDCEmulatorTasks(void);
94 *
95 * Overview: Keeps the demo running.
96 *
97 * PreCondition: The demo should have been initialized and started via
98 * the APP_DeviceCDCEmulatorInitialize() and APP_DeviceCDCEmulatorStart() demos
99 * respectively.
100 *
101 * Input: None
102 *
103 * Output: None
104 *
105 ********************************************************************/
106 void APP_DeviceCDCEmulatorTasks()
107 {
108 
109  if((USBDeviceState < CONFIGURED_STATE)||(USBSuspendControl==1)) return;
110 
111  if (RS232_Out_Data_Rdy == 0) // only check for new USB buffer if the old RS232 buffer is
112  { // empty. This will cause additional USB packets to be NAK'd
113  LastRS232Out = getsUSBUSART(RS232_Out_Data,64); //until the buffer is free.
114  if(LastRS232Out > 0)
115  {
116  RS232_Out_Data_Rdy = 1; // signal buffer full
117  RS232cp = 0; // Reset the current position
118  }
119  }
120 
121  //Check if one or more bytes are waiting in the physical UART transmit
122  //queue. If so, send it out the UART TX pin.
123  if(RS232_Out_Data_Rdy && mTxRdyUSART())
124  {
125  #if defined(USB_CDC_SUPPORT_HARDWARE_FLOW_CONTROL)
126  //Make sure the receiving UART device is ready to receive data before
127  //actually sending it.
128  if(UART_CTS == USB_CDC_CTS_ACTIVE_LEVEL)
129  {
130  USART_putcUSART(RS232_Out_Data[RS232cp]);
131  ++RS232cp;
132  if (RS232cp == LastRS232Out)
133  RS232_Out_Data_Rdy = 0;
134  }
135  #else
136  //Hardware flow control not being used. Just send the data.
137  USART_putcUSART(RS232_Out_Data[RS232cp]);
138  ++RS232cp;
139  if (RS232cp == LastRS232Out)
140  RS232_Out_Data_Rdy = 0;
141  #endif
142  }
143 
144  //Check if we received a character over the physical UART, and we need
145  //to buffer it up for eventual transmission to the USB host.
146  if(mDataRdyUSART() && (NextUSBOut < (CDC_DATA_OUT_EP_SIZE - 1)))
147  {
148  USB_Out_Buffer[NextUSBOut] = USART_getcUSART();
149  ++NextUSBOut;
150  USB_Out_Buffer[NextUSBOut] = 0;
151  }
152 
153  #if defined(USB_CDC_SUPPORT_HARDWARE_FLOW_CONTROL)
154  //Drive RTS pin, to let UART device attached know if it is allowed to
155  //send more data or not. If the receive buffer is almost full, we
156  //deassert RTS.
157  if(NextUSBOut <= (CDC_DATA_OUT_EP_SIZE - 5u))
158  {
159  UART_RTS = USB_CDC_RTS_ACTIVE_LEVEL;
160  }
161  else
162  {
163  UART_RTS = (USB_CDC_RTS_ACTIVE_LEVEL ^ 1);
164  }
165  #endif
166 
167  //Check if any bytes are waiting in the queue to send to the USB host.
168  //If any bytes are waiting, and the endpoint is available, prepare to
169  //send the USB packet to the host.
170  if((USBUSARTIsTxTrfReady()) && (NextUSBOut > 0))
171  {
172  putUSBUSART(&USB_Out_Buffer[0], NextUSBOut);
173  NextUSBOut = 0;
174  }
175 
176  CDCTxService();
177 }
